// GTA RP Browser Script // сцена const scene = new THREE.Scene(); // камера const camera = new THREE.PerspectiveCamera( 75, window.innerWidth / window.innerHeight, 0.1, 1000 ); // рендер const renderer = new THREE.WebGLRenderer(); renderer.setSize(window.innerWidth, window.innerHeight); document.body.appendChild(renderer.domElement); // свет const light = new THREE.DirectionalLight(0xffffff,1); light.position.set(5,10,5); scene.add(light); // земля const groundGeo = new THREE.PlaneGeometry(200,200); const groundMat = new THREE.MeshBasicMaterial({color:0x444444}); const ground = new THREE.Mesh(groundGeo,groundMat); ground.rotation.x = -Math.PI/2; scene.add(ground); // игрок const playerGeo = new THREE.BoxGeometry(1,2,1); const playerMat = new THREE.MeshBasicMaterial({color:0x00ff00}); const player = new THREE.Mesh(playerGeo,playerMat); scene.add(player); camera.position.set(0,5,8); camera.lookAt(player.position); // деньги let money = 1000; // оружие let weapon = "none"; // телефон let phoneOpen = false; // HUD const hud = document.createElement("div"); phoneBtn.innerText="📱 Phone"; phoneBtn.style.position="absolute"; phoneBtn.style.right="20px"; phoneBtn.style.bottom="150px"; document.body.appendChild(phoneBtn); // кнопка оружия const gunBtn = document.createElement("button"); gunBtn.innerText="🔫 Weapon"; gunBtn.style.position="absolute"; gunBtn.style.right="20px"; gunBtn.style.bottom="100px"; document.body.appendChild(gunBtn); // кнопка инвентаря const invBtn = document.createElement("button"); invBtn.innerText="🎒 Inventory"; invBtn.style.position="absolute"; invBtn.style.right="20px"; invBtn.style.bottom="50px"; document.body.appendChild(invBtn); // телефон phoneBtn.onclick = () => { phoneOpen = !phoneOpen;
