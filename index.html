-- ======================================================== -- PROJECT: STANDOFF X GMOD (CORE SYSTEM V2) -- ======================================================== local ProjectSystem = { -- ТАБЛИЦА ЗВАНИЙ (ОТ НОВИЧКА ДО ЛЕГЕНДЫ) -- Каждое звание имеет свой ID и визуальный цвет PlayerRanks = { [1] = {title = "Rookie", color = "#C0C0C0", icon = "🥉"}, -- Самое низкое [2] = {title = "Warrior", color = "#4CAF50", icon = "⚔️"}, [3] = {title = "Veteran", color = "#2196F3", icon = "🎖️"}, [4] = {title = "Master", color = "#9C27B0", icon = "🔮"}, [5] = {title = "Elite", color = "#FF5722", icon = "🔥"}, [6] = {title = "Legend", color = "#FFD700", icon = "🏆"} -- Самое высокое }, -- СИСТЕМА АДМИНИСТРАЦИИ (ПРЕФИКСЫ-КОРОНЫ) StaffGroups = { ["moderator"] = {label = "Модератор", crown = "⚪", desc = "Серебряная корона"}, ["admin"] = {label = "Администратор", crown = "🟡", desc = "Золотая корона"}, ["curator"] = {label = "Куратор", crown = "💎", desc = "Платиновая корона"}, ["owner"] = {label = "Владелец", crown = "👑", desc = "Корона Создателя"} }, -- МАГАЗИН И ВАЛЮТА Currency = { Gold = "G", -- Донат валюта Silver = "S" -- Игровая валюта } } -- 1. ЛОГИКА ОТОБРАЖЕНИЯ НИКНЕЙМОВ И ПРЕФИКСОВ -- Эта функция отвечает за то, что видят другие игроки над головой персонажа function GetPlayerOverheadDisplay(ply) local crown = "" local rankInfo = ProjectSystem.PlayerRanks[ply.rank_id or 1] -- Проверка на наличие административных прав для выдачи короны if ProjectSystem.StaffGroups[ply.group] then crown = ProjectSystem.StaffGroups[ply.group].crown end -- Формат: [Корона] Никнейм [Звание] local displayName = string.format("%s %s [%s]", crown, ply.name, rankInfo.title) return displayName, rankInfo.color end -- 2. МЕНЮ СПАВНА (GARRYS MOD STYLE) -- Включает оружие из Standoff, машины и постройки local SpawnMenu = { Categories = { ["Оружие (Standoff)"] = { {name = "M4 Desertigo", type = "weapon", id = "m4_d"}, {name = "AKR Necromancer", type = "weapon", id = "akr_n"}, {name = "AWP Dragon", type = "weapon", id = "awp_dr"} }, ["Транспорт"] = { {name = "Спорткар", type = "vehicle", id = "car_sport"}, {name = "Пикап", type = "vehicle", id = "car_truck"} }, ["Здания / Пропсы"] = { {name = "Стена бетонная", type = "prop", id = "wall_1"}, {name = "Ящик снабжения", type = "prop", id = "crate_wood"}, {name = "Вышка", type = "prop", id = "tower_sniper"} } } } -- Функция вызова спавна объекта function SpawnObject(ply, category, itemIndex) local item = SpawnMenu.Categories[category][itemIndex] -- Проверка: если это обычный сервер, разрешаем всё. -- Если сервер типа Standoff — спавн пропсов может быть запрещен админом. if ply.server_type == "Standoff" and item.type == "prop" then print("Ошибка: На этом сервере нельзя строить!") return end print(ply.name .. " заспавнил: " .. item.name) -- Здесь вызывается код создания 3D модели в мире end -- 3. СИСТЕМА ДОНАТА И ПОКУПОК function PurchaseItem(ply, price, itemName) if ply.gold_balance >= price then ply.gold_balance = ply.gold_balance - price print(ply.name .. " купил " .. itemName .. ". Остаток: " .. ply.gold_balance .. "G") -- Выдача "плюшки" игроку else print("Недостаточно средств для покупки " .. itemName) end end -- ПРИМЕР ИНИЦИАЛИЗАЦИИ ИГРОКА ПРИ ВХОДЕ function OnPlayerJoined(newPlayer) newPlayer.name = "Player123" newPlayer.gold_balance = 500 newPlayer.rank_id = 1 -- Rookie newPlayer.group = "user" -- По умолчанию обычный игрок -- Если это владелец (проверка по ID) -- newPlayer.group = "owner" print("Добро пожаловать в проект! Твой префикс: " .. GetPlayerOverheadDisplay(newPlayer)) end
